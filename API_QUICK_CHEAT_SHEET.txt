ALL AUTHENTICATION ENDPOINTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Register User:
  POST /api/auth/register/
  {email, password, password_confirm, role:"USER"}

Register Owner:
  POST /api/auth/register-owner/
  {email, password, password_confirm}  (role otomatis OWNER)

Login:
  POST /api/auth/login/
  {email, password}

Get Current User:
  GET /api/auth/me/
  Header: Authorization: Bearer {token}

Refresh Token:
  POST /api/auth/token/refresh/
  {refresh: "token"}

ALL API ENDPOINTS (by role)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PUBLIC (no auth needed):
  GET /api/kategori/                    - List categories
  GET /api/umkm/?status=APPROVED        - List approved UMKM

USER (need auth):
  POST /api/umkm-reviews/               - Create review
  GET /api/auth/me/                     - Get current user

OWNER (need auth):
  POST /api/umkm/                       - Create UMKM
  POST /api/umkm-services/              - Create service
  POST /api/umkm-products/              - Create product
  POST /api/umkm-gallery/               - Upload gallery (form-data!)
  PATCH /api/umkm/{id}/                 - Update UMKM
  DELETE /api/umkm/{id}/                - Delete UMKM

ADMIN (need auth):
  POST /api/kategori/                   - Create category
  POST /api/umkm/{id}/approve/          - Approve UMKM
  POST /api/umkm/{id}/reject/           - Reject UMKM
  GET /api/users/                       - List users
  DELETE /api/users/{id}/               - Delete user

UTILITY:
  POST /api/utils/parse-maps/           - Parse Google Maps URL

COMMON ERRORS & FIXES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

405 Method Not Allowed:
  ❌ Endpoint salah atau tidak ada
  ✅ Cek URL sudah benar: /api/auth/register-owner/ (bukan /register)
  ✅ Pastikan ada /api/ di URL!

200 OK tapi return HTML (bukan JSON):
  ❌ URL lupa /api/ → Nginx route ke frontend SPA
  ✅ Gunakan: https://servpoint.my.id/api/umkm/ (ada /api/)
  ✅ Bukan: https://servpoint.my.id/umkm

400 Bad Request:
  ❌ Body JSON syntax error atau field missing
  ✅ Pastikan Content-Type: application/json
  ✅ Pastikan semua field required ada

401 Unauthorized:
  ❌ Token expired atau tidak ada
  ✅ Copy access_token dari login/register response
  ✅ Format: Authorization: Bearer {token} (ada spasi setelah Bearer)

413 Payload Too Large:
  ❌ File terlalu besar
  ✅ Max size: 10MB per file
